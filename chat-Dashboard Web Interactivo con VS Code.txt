  const tabButtons = document.querySelectorAll('.tab-btn');

  // Eventos para los botones
  tabButtons.forEach(button => {
    button.addEventListener('click', () => {

  // Cargar grÃ¡ficos al cargar la pÃ¡gina
  loadCharts();

function loadCharts() {
  const chartConfigs = [
    { id: 'chartSlide2', file: 'assets/data/slide2.json' },
    { id: 'chartSlide3', file: 'assets/data/slide3.json' },
    { id: 'chartSlide4', file: 'assets/data/slide4.json' },
    { id: 'chartSlide5', file: 'assets/data/slide5.json' },
    { id: 'chartSlide6', file: 'assets/data/slide6.json' }
  ];

  chartConfigs.forEach(config => {
    fetch(config.file)
      .then(response => response.json())
      .then(data => {
        const ctx = document.getElementById(config.id).getContext('2d');
        new Chart(ctx, {
          type: 'bar',
          data: data,
          options: {
            responsive: true,
            maintainAspectRatio: false,
            plugins: {
              legend: { display: false },
              tooltip: { enabled: true }
            },
            scales: {
              y: {
                beginAtZero: true,
                ticks: { precision: 0 }
              }
            }
          }
        });
      })
      .catch(err => console.error(`Error cargando ${config.file}:`, err));

}

document.addEventListener('DOMContentLoaded', () => {
  const tabButtons = document.querySelectorAll('.tab-btn');
  const slides = document.querySelectorAll('.slide');

  function showSlide(slideId) {
    slides.forEach(s => s.classList.remove('active'));
    document.getElementById(`slide${slideId}`).classList.add('active');
    tabButtons.forEach(btn => btn.classList.remove('active'));
    document.querySelector(`[data-slide="${slideId}"]`).classList.add('active');
  }

  tabButtons.forEach(button => {
    button.addEventListener('click', () => {
      const slideId = button.getAttribute('data-slide');
      showSlide(slideId);
    });
  });

  loadCharts();
});

function loadCharts() {
  const chartConfigs = [
    { id: 'chartSlide2', file: 'assets/data/slide2.json' },
    { id: 'chartSlide3', file: 'assets/data/slide3.json' },
    { id: 'chartSlide4', file: 'assets/data/slide4.json' },
    { id: 'chartSlide5', file: 'assets/data/slide5.json' },
    { id: 'chartSlide6', file: 'assets/data/slide6.json' }
  ];

  chartConfigs.forEach(config => {
    fetch(config.file)
      .then(response => response.json())
      .then(data => {
        const ctx = document.getElementById(config.id).getContext('2d');
        new Chart(ctx, {
          type: 'bar',
           data,
          options: {
            responsive: true,
            maintainAspectRatio: false,
            plugins: {
              legend: { display: false },
              tooltip: { enabled: true }
            },
            scales: {
              y: {
                beginAtZero: true,
                ticks: { precision: 0 }
              }
            }
          }
        });
      })
      .catch(err => console.error(`Error cargando ${config.file}:`, err));
  });
}

document.addEventListener('DOMContentLoaded', () => {
  const tabButtons = document.querySelectorAll('.tab-btn');
  const slides = document.querySelectorAll('.slide');

  function showSlide(slideId) {
    slides.forEach(s => s.classList.remove('active'));
    document.getElementById(`slide${slideId}`).classList.add('active');
    tabButtons.forEach(btn => btn.classList.remove('active'));
    document.querySelector(`[data-slide="${slideId}"]`).classList.add('active');
  }

  tabButtons.forEach(button => {
    button.addEventListener('click', () => {
      const slideId = button.getAttribute('data-slide');
      showSlide(slideId);
    });
  });

  loadCharts();
});

function loadCharts() {
  const chartConfigs = [
    { id: 'chartSlide2', file: 'assets/data/slide2.json', type: 'bar' },
    { id: 'chartSlide3', file: 'assets/data/slide3.json', type: 'bar' },
    { id: 'chartSlide4', file: 'assets/data/slide4.json', type: 'line' },
    { id: 'chartSlide5', file: 'assets/data/slide5.json', type: 'bar' },
    { id: 'chartSlide6', file: 'assets/data/slide6.json', type: 'bar' }
  ];

  chartConfigs.forEach(config => {
    fetch(config.file)
      .then(response => response.json())
      .then(data => {
        const ctx = document.getElementById(config.id).getContext('2d');
        new Chart(ctx, {
          type: config.type,
           data,
          options: {
            responsive: true,
            maintainAspectRatio: false,
            animation: { duration: 0 }, // â¬…ï¸ SIN ANIMACIÃ“N
            plugins: {
              legend: { display: false },
              tooltip: { enabled: true }
            },
            scales: {
              y: {
                beginAtZero: true,
                ticks: { precision: 0 }
              }
            }
          }
        });
      })
      .catch(err => console.error(`Error cargando ${config.file}:`, err));
  });
}
